Quick Start

// Initialize the shading system
const shading = new MetallicShadingSystem();

// Generate a 5-stop metallic gradient
const gradient = shading.generateMetallicGradient('#b829dd', 1.2);

// Calculate surface shading
const shading = shading.calculateSurfaceShading(45, {x: 0.3, y: -0.5});

// Apply dynamic color based on price change
const color = shading.applyDynamicColor('#b829dd', 5.42);

// Create SVG ribbon gradient
const svgGradient = shading.createRibbonGradient(gradient, null, 'ribbon-grad');

// Create complete metallic ribbon SVG
const svg = shading.createMetallicRibbon(300, 60, '#b829dd', 5.42);
